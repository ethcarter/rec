#### **Aim of the experiment:**
> Write a program using C++ decision-making statements and branching statements

---

#### **Program:**
> **Decision Making**
```cpp
#include <iostream>
using namespace std;

int main() {
    int num;
    cout << "Enter a number: ";
    cin >> num;

    if (num > 0)
        cout << "The number is Positive" << endl;
    else if (num < 0)
        cout << "The number is Negative" << endl;
    else
        cout << "The number is Zero" << endl;

    return 0;
}
```
<br>

> **Branching**
```cpp
#include <iostream>
using namespace std;

int main() {
    char ch;
    cout << "Enter a single alphabet character: ";
    cin >> ws >> ch; // Ignore leading whitespace

    switch (ch) {
        case 'a': case 'A':
        case 'e': case 'E':
        case 'i': case 'I':
        case 'o': case 'O':
        case 'u': case 'U':
            cout << "'" << ch << "' is a vowel." << endl;
            break;
        default:
            // Check if input is an alphabet character
            if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')) {
                cout << "'" << ch << "' is a consonant." << endl;
            } else {
                cout << "'" << ch << "' is not an alphabet character." << endl;
            }
    }

    return 0;
}
```
