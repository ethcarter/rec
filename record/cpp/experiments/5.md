#### **Aim of the experiment:**
> Write a program using *Loop* statements in C++

---

#### **Program:**
```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter a number: ";
    cin >> n;

    // For loop
    cout << "\nUsing for loop:\n";
    for (int i = 1; i <= n; i++) {
        cout << i << " ";
    }
    cout << endl;

    // While loop
    cout << "\nUsing while loop:\n";
    int i = 1;
    while (i <= n) {
        cout << i << " ";
        i++;
    }
    cout << endl;

    // Do-while loop
    cout << "\nUsing do-while loop:\n";
    i = 1;
    if (n >= 1) {
        do {
            cout << i << " ";
            i++;
        } while (i <= n);
    }
    cout << endl;

    return 0;
}
```
