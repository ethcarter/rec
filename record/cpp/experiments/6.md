#### **Aim of the experiment:**
> Write a program to demonstrate applications of n-dimensional arrays in C++

---

#### **Program:**
```cpp
#include <iostream>
using namespace std;

int main() {
    // 2D Array - Matrix operations
    int matrix[3][3] = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
    
    cout << "2D Array (Matrix):" << endl;
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            cout << matrix[i][j] << " ";
        }
        cout << endl;
    }
    
    // 3D Array - Student marks
    int marks[2][3][2] = {{{80, 85}, {90, 88}, {75, 82}},
                          {{88, 92}, {85, 87}, {90, 95}}};
    
    cout << "\n3D Array (Student Marks):" << endl;
    for (int i = 0; i < 2; i++) {
        cout << "Class " << i + 1 << ":" << endl;
        for (int j = 0; j < 3; j++) {
            cout << "Student " << j + 1 << ": ";
            for (int k = 0; k < 2; k++) {
                cout << marks[i][j][k] << " ";
            }
            cout << endl;
        }
    }
    
    return 0;
}
```
