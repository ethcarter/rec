EXPERIMENT 09

C++ Programming

Aim of the experiment:

Write a program in C++ to demonstrate recursion

Program:

#include <iostream>
using namespace std;

int factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

int fibonacci(int n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

void printNumbers(int n) {
    if (n > 0) {
        cout << n << " ";
        printNumbers(n - 1);
    }
}

int main() {
    int num = 5;
    
    cout << "Factorial of " << num << ": " << factorial(num) << endl;
    
    cout << "Fibonacci series (first 6 terms): ";
    for (int i = 0; i < 6; i++) {
        cout << fibonacci(i) << " ";
    }
    cout << endl;
    
    cout << "Numbers from " << num << " to 1: ";
    printNumbers(num);
    cout << endl;
    
    return 0;
}

Sample Output:

Factorial of 5: 120
Fibonacci series (first 6 terms): 0 1 1 2 3 5 
Numbers from 5 to 1: 5 4 3 2 1

Conclusion:

Recursion elegantly solves problems by breaking them into smaller subproblems with base and recursive cases
