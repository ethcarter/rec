#### **Aim of the experiment:**
> Implement graph representations (adjacency list, adjacency
matrix) and basic graph traversals (BFS, DFS)

#### **Program:**
```c
#include <stdio.h>
#include <stdlib.h>

#define MAX 10

// Adjacency Matrix Representation
int adjMatrix[MAX][MAX];
int visited[MAX];

// Adjacency List Node
struct Node {
    int data;
    struct Node* next;
};

struct Node* adjList[MAX];

// Queue for BFS
int queue[MAX], front = -1, rear = -1;

// Stack for DFS
int stack[MAX], top = -1;

void addEdgeMatrix(int src, int dest) {
    adjMatrix[src][dest] = 1;
    adjMatrix[dest][src] = 1; // For undirected graph
}

void addEdgeList(int src, int dest) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = dest;
    newNode->next = adjList[src];
    adjList[src] = newNode;
    
    // For undirected graph
    newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = src;
    newNode->next = adjList[dest];
    adjList[dest] = newNode;
}

void enqueue(int item) {
    if (rear == MAX - 1) return;
    if (front == -1) front = 0;
    queue[++rear] = item;
}

int dequeue() {
    if (front == -1) return -1;
    int item = queue[front++];
    if (front > rear) front = rear = -1;
    return item;
}

void push(int item) {
    if (top == MAX - 1) return;
    stack[++top] = item;
}

int pop() {
    if (top == -1) return -1;
    return stack[top--];
}

void resetVisited() {
    for (int i = 0; i < MAX; i++) {
        visited[i] = 0;
    }
}

void bfsMatrix(int start, int vertices) {
    resetVisited();
    enqueue(start);
    visited[start] = 1;
    
    printf("BFS (Matrix): ");
    while (front != -1) {
        int current = dequeue();
        printf("%d ", current);
        
        for (int i = 0; i < vertices; i++) {
            if (adjMatrix[current][i] == 1 && !visited[i]) {
                enqueue(i);
                visited[i] = 1;
            }
        }
    }
    printf("\n");
}

void dfsMatrix(int start, int vertices) {
    resetVisited();
    push(start);
    
    printf("DFS (Matrix): ");
    while (top != -1) {
        int current = pop();
        if (!visited[current]) {
            printf("%d ", current);
            visited[current] = 1;
            
            for (int i = vertices - 1; i >= 0; i--) {
                if (adjMatrix[current][i] == 1 && !visited[i]) {
                    push(i);
                }
            }
        }
    }
    printf("\n");
}

void bfsList(int start) {
    resetVisited();
    enqueue(start);
    visited[start] = 1;
    
    printf("BFS (List): ");
    while (front != -1) {
        int current = dequeue();
        printf("%d ", current);
        
        struct Node* temp = adjList[current];
        while (temp) {
            if (!visited[temp->data]) {
                enqueue(temp->data);
                visited[temp->data] = 1;
            }
            temp = temp->next;
        }
    }
    printf("\n");
}

void dfsList(int start) {
    resetVisited();
    push(start);
    
    printf("DFS (List): ");
    while (top != -1) {
        int current = pop();
        if (!visited[current]) {
            printf("%d ", current);
            visited[current] = 1;
            
            struct Node* temp = adjList[current];
            while (temp) {
                if (!visited[temp->data]) {
                    push(temp->data);
                }
                temp = temp->next;
            }
        }
    }
    printf("\n");
}

void printMatrix(int vertices) {
    printf("Adjacency Matrix:\n");
    for (int i = 0; i < vertices; i++) {
        for (int j = 0; j < vertices; j++) {
            printf("%d ", adjMatrix[i][j]);
        }
        printf("\n");
    }
}

void printList(int vertices) {
    printf("Adjacency List:\n");
    for (int i = 0; i < vertices; i++) {
        printf("%d: ", i);
        struct Node* temp = adjList[i];
        while (temp) {
            printf("%d -> ", temp->data);
            temp = temp->next;
        }
        printf("NULL\n");
    }
}

int main() {
    int vertices = 5;
    
    // Initialize
    for (int i = 0; i < MAX; i++) {
        for (int j = 0; j < MAX; j++) {
            adjMatrix[i][j] = 0;
        }
        adjList[i] = NULL;
    }
    
    // Add edges
    addEdgeMatrix(0, 1);
    addEdgeMatrix(0, 2);
    addEdgeMatrix(1, 3);
    addEdgeMatrix(2, 4);
    
    addEdgeList(0, 1);
    addEdgeList(0, 2);
    addEdgeList(1, 3);
    addEdgeList(2, 4);
    
    printMatrix(vertices);
    printf("\n");
    printList(vertices);
    printf("\n");
    
    bfsMatrix(0, vertices);
    dfsMatrix(0, vertices);
    bfsList(0);
    dfsList(0);
    
    return 0;
}
```
